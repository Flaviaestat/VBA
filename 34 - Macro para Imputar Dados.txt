Sub IMPUTA_DADOS()

Dim Linha As Range


For Each Linha In Sheets("Input").Range("$B$2:$B$22")
Sheets("Conexão ODBC").Activate
Range("A:B").ClearContents
Entrada = Linha.Value

   With ActiveSheet.ListObjects.Add(SourceType:=0, Source:= _
        "ODBC;DSN=RJD10;UID=RERAZO;PWD=RERAZO;DBQ=RJD10;ASY=OFF;", Destination:=Range("$A$1")). _
        QueryTable
        .CommandText = Array("INSERT INTO TESTE_FLA (VARIÁVEL) VALUES('" & Entrada & "')")
        .CommandType = xlCmdSql
        .RowNumbers = False
        .FillAdjacentFormulas = False
        .PreserveFormatting = True
        .RefreshOnFileOpen = False
        .BackgroundQuery = True
        .RefreshStyle = xlInsertDeleteCells
        .SavePassword = False
        .SaveData = True
        .AdjustColumnWidth = True
        .RefreshPeriod = 0
        .PreserveColumnInfo = True
        End With
        
        
ActiveWorkbook.RefreshAll

Sheets("Resultado").Activate
Range("A:B").ClearContents

With ActiveSheet.ListObjects.Add(SourceType:=0, Source:= _
        "ODBC;DSN=RJD10;UID=RERAZO;PWD=RERAZO;DBQ=RJD10;ASY=OFF;", Destination:=Range("$A$1")). _
        QueryTable
        .CommandText = Array("SELECT * FROM TESTE_FLA")
        .CommandType = xlCmdSql
        .RowNumbers = False
        .FillAdjacentFormulas = False
        .PreserveFormatting = True
        .RefreshOnFileOpen = False
        .BackgroundQuery = True
        .RefreshStyle = xlInsertDeleteCells
        .SavePassword = False
        .SaveData = True
        .AdjustColumnWidth = True
        .RefreshPeriod = 0
        .PreserveColumnInfo = True
        .Refresh BackgroundQuery:=False
        .ListObject.DisplayName = "Tabela_Consulta_de_RJD10"
        End With

Next Linha


End Sub
